  addId();

function addDiscount() {
  medicines.map((medicine) => {
    if (medicine.price >= 300) {
      medicine.price = medicine.price - (medicine.price * 0.3);
    }
  });
}

function addId() {

Додай методи класу:
getItems() - повертайте масив товарів
addItems(item) - отримує новий товар та додає його до поточних

const arr = ["apple", "banana", "mango"]

Виконання 
class Storage {
  constructor(items) {
    this.items = items;
  }
  getItems() {
    return this.items;
  }
  addItems(item) {
    this.items.push(item);
  }
  removeItem(item) {
    this.items = this.items.filter((i) => i !== item);
  }
}

    { id: "002", likes: 17, tags: ["html", "js", "nodejs"] },

Виконання 
function countTags() { 
  const tags = new Map();

  for (const tweet of tweets) {
    for (const tag of tweet.tags) {
      tags.set(tag, (tags.get(tag) || 0) + 1);
    }
  }

  return tags;
}

Постановка задачі

Напишіть функцію checkBrackets(str) яка приймає рядок js коду (someFn)  і перевіряє правильність закриття дужок () {} []
Якщо рядок містить коректний код функція повертає true.
В іншому випадку повертає false

Виконання 

function checkBrackets(str) {
  const openBrackets = new Set(["(", "{", "["]);
  const closeBrackets = new Set([")", "}", "]"])
  const bracketsPairs = {
    ")": "(",
    "}": "{",
    "]" : "[",
  };

  let stack = [];

  for (const char of str) {
    if (openBrackets.has(char)) 
    {
      stack.push(char);
    } 
    else if (closeBrackets.has(char)) 
    {
      if(stack.length === 0)
        return false;

      if(bracketsPairs[char] != stack.pop())
        return false;
    }
  }

  return stack.length === 0;
};

Об'єкти в JavaScript - це основні структури даних, які використовуються для зберігання даних та функціональних можливостей. Вони являють собою набори пар ключ-значення, де ключ - це строка, а значення - це будь-який тип даних, включаючи числа, строки, масиви, функції тощо.

Ось приклад об'єкта: 

const person = {
  name: "John Doe",
  age: 30,
  occupation: "Software Engineer",
};

Цей об'єкт має три властивості:
  • name: строка, яка містить ім'я людини
  • age: число, яке містить вік людини
  • occupation: строка, яка містить професію людини
Об'єкти можна оголошувати за допомогою літералу об'єкта, як показано вище, або за допомогою конструктора об'єкта: 
const person = new Object();
person.name = "John Doe";
person.age = 30;
person.occupation = "Software Engineer";
Об'єкти можна використовувати для представлення різних типів даних. Наприклад, можна використовувати об'єкт для представлення людини, тварини, продукту або навіть веб-сторінки.

Стрілочні функції в JavaScript - це скорочений варіант оголошення функції. Вони були введені в стандарті ES6 і мають ряд переваг перед звичайними функціями.
Ось синтаксис стрілочної функції: 

(параметри) => вираз

Наприклад, наступні два фрагменти коду є рівнозначними: 
function add(a, b) {
  return a + b;
}

const add = (a, b) => a + b;

Стрілочні функції мають такі переваги перед звичайними функціями:
  • Коротший синтаксис: Стрілочні функції мають коротший синтаксис, ніж звичайні функції. Наприклад, стрілочна функція (a, b) => a + b замінює звичайну функцію function add(a, b) { return a + b; }.

  • Легше читати і розуміти: Стрілочні функції простіше читати і розуміти, ніж звичайні функції. Наприклад, стрілочна функція (a, b) => a + b зрозуміла відразу, що вона повертає суму двох чисел.

  • Не мають власного об'єкта arguments: Стрілочні функції не мають власного об'єкта arguments. Це означає, що стрілочні функції не можуть отримати доступ до аргументів, які були передані в них, за допомогою об'єкта arguments.

  • Виконуються в контексті, в якому вони були оголошені: Стрілочні функції виконуються в контексті, в якому вони були оголошені. Це означає, що стрілочні функції мають доступ до змінних і функцій, які були визначені в цьому контексті.
